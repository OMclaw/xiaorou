**优化方案**:
```python
def get_optimal_size(reference_image: Path) -> str:
    """根据参考图宽高比自动选择分辨率"""
    from PIL import Image
    img = Image.open(reference_image)
    w, h = img.size
    ratio = w / h
    
    if ratio > 1.3:  # 横版
        return '2048*1536'  # 16:9
    elif ratio < 0.8:  # 竖版
        return '1536*2048'  # 3:4
    else:  # 正方形
        return '1792*1792'  # 1:1

# 在 generate_role_swap_image() 中调用
size = get_optimal_size(reference_image_path)
payload['parameters']['size'] = size
```

#### 3. 智能重试机制

**现状**:
```python
for attempt in range(max_retries + 1):
    # 固定重试 3 次
    # 固定等待时间 time.sleep(1 * (attempt + 1))
```

**问题**:
- 固定重试次数不够灵活
- 固定等待时间未考虑 API 限流
- 未区分错误类型（审核失败 vs 网络错误）

**优化方案**:
```python
def generate_with_smart_retry(...):
    max_retries = 5
    base_delay = 2
    
    for attempt in range(max_retries + 1):
        try:
            response = requests.post(...)
            
            if response.status_code == 429:  # 限流
                retry_after = int(response.headers.get('Retry-After', base_delay * (attempt + 1)))
                logger.warning(f"⚠️ API 限流，等待{retry_after}秒后重试...")
                time.sleep(retry_after)
                continue
                
            result_json = response.json()
            error_code = result_json.get('code', '')
            
            if error_code == 'DataInspectionFailed':  # 审核失败
                logger.error("❌ 内容审核失败，不建议重试")
                return (model_name, None)  # 直接返回失败
                
            elif error_code == 'InvalidParameter':  # 参数错误
                # 自动调整参数后重试
                payload['parameters']['prompt_extend'] = True
                logger.warning("⚠️ 参数无效，已调整参数后重试...")
                continue
                
            elif response.status_code == 200 and result_json.get('output'):
                # 成功
                return (model_name, image_url)
                
        except requests.Timeout:
            # 超时错误，增加等待时间
            delay = base_delay * (2 ** attempt)  # 指数退避
            logger.warning(f"⚠️ 请求超时，等待{delay}秒后重试...")
            time.sleep(delay)
            continue
```

**预期提升**:
- 成功率 +8%
- API 限流应对 +15%
- 减少无效重试

```python
def analyze_reference_image(image_path: Path) -> dict:
    """深度分析参考图，提取关键信息"""
    from PIL import Image
    import cv2
    import numpy as np
    
    img = Image.open(image_path)
    analysis = {
        'resolution': f"{img.width}x{img.height}",
        'aspect_ratio': img.width / img.height,
        'dominant_colors': extract_dominant_colors(img),  # 提取主色调
        'lighting_direction': analyze_lighting(img),  # 分析光源方向
        'scene_type': classify_scene(img),  # 场景分类（室内/室外/泳池/海边等）
        'clothing_type': classify_clothing(img),  # 服装分类（泳装/连衣裙/休闲装等）
        'pose_type': classify_pose(img),  # 姿势分类（站姿/坐姿/躺姿等）
        'has_watermark': detect_watermark(img),  # 水印检测
        'face_angle': estimate_face_angle(img),  # 脸部角度估计
        'quality_score': assess_image_quality(img),  # 画质评分
    }
    
    # 根据分析结果自动调整 prompt
    if analysis['scene_type'] == '泳池':
        prompt_additions.append("清澈泳池背景，水面反射")
    if analysis['lighting_direction'] == '侧光':
        prompt_additions.append("侧面光源，明暗对比明显")
    if analysis['has_watermark']:
        prompt_additions.append("(无水印：6.0)，(去除参考图水印：6.0)")
    
    return analysis
```

**预期提升**:
