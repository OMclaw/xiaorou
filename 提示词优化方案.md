### Step 1: 创建结构化 Prompt 构建函数

```python
def build_structured_prompt(reference_description: str = "") -> str:
    """
    构建结构化角色替换 prompt - v7.0 优化版
    
    优化点:
    1. 消除重复内容（减少 30% 长度）
    2. 统一权重标注（所有 5.0+ 项集中）
    3. 中英文分离（避免混排）
    4. 关键指令前置（提升注意力）
    """
    
    # ========== 核心指令（前置，最高优先级）==========
    core_instruction = """【核心指令 - 最高优先级】
任务：角色替换（图 1 场景 + 图 2 脸）
图 1（参考图）：场景/服装/姿势/光影/构图 → 人脸完全忽略
图 2（小柔）：脸部/五官/发型/肤色 → 100% 完全保留
"""
    
    # ========== 必须保持的内容 ==========
    must_keep = """【必须保持 - 图 1 内容】
✅ 场景背景、服装穿搭、姿势动作
✅ 光源方向/强度/色温、阴影位置
✅ 构图镜头、景深效果、整体色调
"""
    
    # ========== 必须忽略的内容 ==========
    must_ignore = """【必须忽略 - 图 1 内容】
❌ 人脸（完全覆盖替换）
❌ 发型、肤色（使用图 2）
❌ 所有水印、文字、logo（完全去除）
"""
    
    # ========== 质量标准 ==========
    quality_standards = """【质量标准】
• 头身比 1:8-1:9，小脸效果
• 脸部/颈部/身体肤色 100% 一致，无色差
• 边缘融合自然，下巴颈部无分界线
• 真实摄影质感，无 AI 感/塑料感
"""
    
    # ========== 反向提示词（权重分级）==========
    negative_prompts = """【反向提示词 - 权重分级】
权重 6.0（绝对禁止）:
(watermark:6.0), (text:6.0), (logo:6.0), (no watermark:6.0)

权重 5.0（高优先级）:
(plastic skin:5.0), (ai skin:5.0), (face neck seam:5.0),
(wrong shadow:5.0), (wrong lighting:5.0), (wrong face angle:5.0),
(big head:5.0), (head body mismatch:5.0), (expression mismatch:5.0)

权重 4.0（中优先级）:
(worst quality, low quality:1.4), (deformed, distorted:1.3), bad anatomy,
(wrong depth of field:4.0), (wrong noise:4.0), (perspective conflict:4.0)
"""
    
    # ========== 关键英文指令（集中放置）==========
    critical_english = """【CRITICAL INSTRUCTIONS】
FACE LOCK: 100% USE image-2 face, IGNORE image-1 face COMPLETELY
SKIN TONE: face/neck/body color must be 100% unified
LIGHTING: match light direction/shadows/highlights/color temperature
WATERMARK: ABSOLUTELY NO WATERMARK - REMOVE ALL completely
PROPORTION: correct head-body ratio 1:8, small face effect
FUSION: natural edge blending, no visible seams
"""
    
    # ========== 参考图描述（动态内容）==========
    ref_description = f"""【参考图描述】{reference_description if reference_description else "标准人像场景"}"""
    
    # ========== 组装完整 Prompt ==========
    full_prompt = "\n".join([
        core_instruction,
        must_keep,
        must_ignore,
        ref_description,
        quality_standards,
        negative_prompts,
        critical_english,
    ])
    
    return full_prompt
```

### Step 2: 优化权重标注系统

```python
# 当前：分散在各处，权重不统一
"""
(无水印：5.0)  # 在 instruction 中
(无水印：5.0)  # 在无水印要求中
(无水印：5.0)  # 在 EXTREMELY CRITICAL 中
"""

# 优化后：集中管理，权重分级
WEIGHT_CRITICAL = 6.0  # 绝对禁止（水印/文字/logo）
WEIGHT_HIGH = 5.0      # 高优先级（人脸/肤色/光影）
WEIGHT_MEDIUM = 4.0    # 中优先级（画质/透视）
WEIGHT_LOW = 1.4       # 基础质量（低质量/畸形）

negative_prompts = {
    WEIGHT_CRITICAL: ["watermark", "text", "logo", "no watermark"],
    WEIGHT_HIGH: ["plastic skin", "face neck seam", "wrong shadow", 
                  "wrong lighting", "big head", "head body mismatch"],
    WEIGHT_MEDIUM: ["wrong depth of field", "wrong noise", "perspective conflict"],
    WEIGHT_LOW: ["worst quality", "low quality", "deformed", "distorted"],
}

def build_negative_prompt():
    """构建分级反向提示词"""
    parts = []
    for weight, items in negative_prompts.items():
        for item in items:
            parts.append(f"({item}:{weight})")
    return ", ".join(parts)
```

### Step 4: Prompt 长度优化

```python
def optimize_prompt_length(prompt: str, max_length: int = 5000) -> str:
    """智能截断 prompt，保留关键信息"""
    
    if len(prompt) <= max_length:
        return prompt
    
    # 优先级：核心指令 > 质量标准 > 反向提示词 > 参考图描述
    sections = prompt.split("\n\n")
    priority_order = {
        "核心指令": 1,
        "必须保持": 2,
        "必须忽略": 3,
        "质量标准": 4,
        "反向提示词": 5,
        "CRITICAL": 6,
        "参考图描述": 7,
    }
    
    # 按优先级排序
    sections.sort(key=lambda s: priority_order.get(s.split("【")[1].split("】")[0], 99))
    
    # 从低优先级开始截断
    result = []
    current_length = 0
    for section in sections:
        if current_length + len(section) + 2 <= max_length:
            result.append(section)
            current_length += len(section) + 2
        else:
            # 部分保留参考图描述
            if "参考图描述" in section:
                result.append(section[:max_length - current_length - 10] + "...")
            break
    
    return "\n\n".join(result)
```
